```bash

  --name workspace-portal \
  -p 3000:3000 \
  -e PORTAL_USER=admin \
  -e PORTAL_PASS=yourpassword \
  -v /root/.openclaw/workspace:/root/.openclaw/workspace \
  su600/workspace-portal:latest
```
